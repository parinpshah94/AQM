Part 2

# Sample from 2 normal distributions
x1 <- rnorm(100)
x2 <- rnorm(100)

# Construct constants
beta0 <- 0
beta1 <- 2
beta2 <- 8

# Construct equation
z <- beta0 + beta1*x1 + beta2*x2 + rnorm(100)

We regress using Rs `lm function based on the above equation

## 
## Call:
## lm(formula = z ~ x1 + x2 + rnorm(100))
## 
## Coefficients:
## (Intercept)           x1           x2   rnorm(100)  
##     0.07112      1.93631      7.92040      0.11848
# Build a matrix from above
mat <- cbind(z, x1, x2)
# Examine plots
pairs(mat)
